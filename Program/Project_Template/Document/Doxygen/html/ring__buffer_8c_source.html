103 uint8_t Ring_Buffer_Write_String(ring_buffer *ring_buffer_handle, void *input_data, uint32_t write_lenght)
104 {
105  //如果不够存储空间存放新数据,返回错误
106  if((ring_buffer_handle->lenght + write_lenght) > (ring_buffer_handle->max_lenght))
107  {
108  return RING_BUFFER_ARRAY_ERROR ;
109  }
110  else
111  {
112  //设置两次写入长度
113  uint32_t write_size_a, write_size_b ;
114  //如果顺序可用长度小于需写入的长度，需要将数据拆成两次分别写入
115  if((ring_buffer_handle->max_lenght - ring_buffer_handle->tail) < write_lenght)
116  {
117  write_size_a = ring_buffer_handle->max_lenght - ring_buffer_handle->tail ;//从尾指针开始写到储存数组末尾
118  write_size_b = write_lenght - (ring_buffer_handle->max_lenght - ring_buffer_handle->tail) ;//从储存数组开头写数据
119  }
120  else//如果顺序可用长度大于需写入的长度，则只需要写入一次
121  {
122  write_size_a = write_lenght ;//从尾指针开始写到储存数组末尾
123  write_size_b = 0 ;//无需从储存数组开头写数据
124  }
125  //开始写入数据
126  if(write_size_b != 0)//需要写入两次
127  {
128  //分别拷贝a、b段数据到储存数组中
129  memcpy(ring_buffer_handle->array_addr + ring_buffer_handle->tail, input_data, write_size_a);
130  memcpy(ring_buffer_handle->array_addr, input_data + write_size_a , write_size_b);
131  ring_buffer_handle->lenght += write_lenght ;//记录新存储了多少数据量
132  ring_buffer_handle->tail = write_size_b ;//重新定位尾指针位置
133  }
134  else//只需写入一次
135  {
136  memcpy(ring_buffer_handle->array_addr + ring_buffer_handle->tail, input_data, write_size_a);
137  ring_buffer_handle->lenght += write_lenght ;//记录新存储了多少数据量
138  ring_buffer_handle->tail += write_size_a ;//重新定位尾指针位置
139  }
140  return RING_BUFFER_ARRAY_SUCCESS ;
141  }
